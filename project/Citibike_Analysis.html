  #Calculate distance using Euclidean Distance formula
  df_param['distance'] = (df_param['lat']-new_latitude)**2 + (df_param['lon']-new_longitude)**2
  N = len(df_param['distance'])
  for i in range(N):
    if df_param['num_bikes_available'][i]<=0:
      df_param['distance'][i] = 9999
  return df_param.loc[df_param['distance'].idxmin()]
